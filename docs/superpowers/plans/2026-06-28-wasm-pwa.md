# Arkanoid WASM PWA Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Compile `Arkanoid.Core` to browser WebAssembly, replace the WebSocket+REST transport with JS↔WASM in-process calls, and ship a fully static PWA playable on mobile with no server.

**Architecture:** A new `Arkanoid.Wasm` C# project (target `net8.0-browser`) replaces `Arkanoid.Server` — it exposes game session lifecycle and all meta operations via `[JSExport]`, stores profile+dungeon state in `localStorage` via `[JSImport]`, and bundles all config JSONs as embedded resources. The frontend gains `WasmConnection.ts` (game loop) and `WasmApi.ts` (meta calls) that mirror the existing `Connection`/`metaApi` interfaces exactly, so `BattleScene.svelte` and all scene files are untouched. The build pipeline runs `dotnet publish` on the Wasm project then copies the `_framework/` output into Vite's public folder; the final `dist/` is a self-contained static site.

**Tech Stack:** .NET 8 browser-wasm, `System.Runtime.InteropServices.JavaScript` ([JSExport]/[JSImport]), TypeScript, Svelte 5, Vite, Pixi.js, PWA manifest + service worker.

**Separation preserved:** `Arkanoid.Core` = pure game engine (no transport, no HTTP). `Arkanoid.Wasm` = browser transport layer (same role as `Arkanoid.Server` but compiled to WASM). Frontend calls through typed TS interfaces — it never knows whether it is talking to a WebSocket or a WASM module.

using System.Reflection;
using System.Text;

namespace ArkanoidWasm;

internal static class ResourceLoader
{
    private static readonly Assembly _asm = typeof(ResourceLoader).Assembly;

    internal static string GetJson(string logicalName)
    {
        var stream = _asm.GetManifestResourceStream(logicalName)
            ?? throw new InvalidOperationException($"Embedded resource not found: {logicalName}");
        using var reader = new StreamReader(stream, Encoding.UTF8);
        return reader.ReadToEnd();
    }

    internal static bool TryGetJson(string logicalName, out string json)
    {
        var stream = _asm.GetManifestResourceStream(logicalName);
        if (stream == null) { json = ""; return false; }
        using var reader = new StreamReader(stream, Encoding.UTF8);
        json = reader.ReadToEnd();
        return true;
    }

    internal static IEnumerable<string> ListLevelIds()
    {
        return _asm.GetManifestResourceNames()
            .Where(n => n.StartsWith("config/levels/") && n.EndsWith(".json"))
            .Select(n => n["config/levels/".Length..].Replace(".json", ""));
    }
}
```

using System.Text.Json;
using Arkanoid.Core.Meta;

namespace ArkanoidWasm;

internal sealed class WasmProfileStore : IProfileStore
{
    private static readonly JsonSerializerOptions _opts = new() { WriteIndented = false };
    private const string KeyPrefix = "ark_profile_";

    public Profile Load(string pid = "default")
    {
        var key = KeyPrefix + Sanitize(pid);
        var json = StorageInterop.Get(key);
        if (string.IsNullOrEmpty(json)) return Profile.NewDefault();
        var p = JsonSerializer.Deserialize<Profile>(json) ?? Profile.NewDefault();
        if (!p.CurrencyMigrated)
        {
            p.MigrateCurrencies();
            Save(p, pid);
        }
        return p;
    }

    public void Save(Profile profile, string pid = "default")
    {
        var key = KeyPrefix + Sanitize(pid);
        StorageInterop.Set(key, JsonSerializer.Serialize(profile, _opts));
    }

    private static string Sanitize(string s) =>
        new string((s ?? "").Where(c => char.IsLetterOrDigit(c) || c is '-' or '_').ToArray()) is { Length: > 0 } v ? v : "default";
}
```

```csharp
using System.Runtime.InteropServices.JavaScript;
using System.Text.Json;
using Arkanoid.Core;
using Arkanoid.Core.Blocks;
using Arkanoid.Core.Bonuses;
using Arkanoid.Core.Net;
using Arkanoid.Core.Relics;
using Arkanoid.Core.Sim;

namespace ArkanoidWasm;

public static partial class GameBridge
{
    private static GameInstance? _game;
    private static long _tick;
    private static readonly Dictionary<string, string[]> _blockCache = new();
    private static readonly Queue<InputCommand> _inbox = new();

    // Catalogs loaded once at module init (Program.cs runs first).
    private static BlockCatalog? _blockCatalog;
    private static RelicCatalog? _relicCatalog;
    private static BonusCatalog? _bonusCatalog;
    private static readonly WasmProfileStore _profileStore = new();
    private static readonly WasmDungeonStore _dungeonStore = new();

    // Called once per page load (from bridge.ts after dotnet.create()).
    [JSExport]
    public static void InitCatalogs()
    {
        _blockCatalog  = BlockCatalog.FromJson(ResourceLoader.GetJson("config/blocks.json"));
        _relicCatalog  = RelicCatalog.FromJson(ResourceLoader.GetJson("config/relics.json"));
        _bonusCatalog  = ResourceLoader.TryGetJson("config/bonuses.json", out var bj)
                         ? BonusCatalog.FromJson(bj) : null;
    }

    // Called by WasmConnection constructor (one per battle scene).
    [JSExport]
    public static void InitSession(string levelId, int seed, string runId, string pid, string mode)
    {
        _blockCache.Clear();
        _inbox.Clear();
        _tick = 0;

        // Replicate the trial-mode level/seed override from GameSession.
        if (mode == "trial")
        {
            var weekId = Arkanoid.Core.Meta.SeasonClock.Default.WeekId(DateTimeOffset.UtcNow);
            levelId = Arkanoid.Core.Meta.TrialConfig.LevelId;
            seed    = Arkanoid.Core.Meta.TrialConfig.SeedFor(weekId);
        }

        // NullSimLog: WASM has no file system to write logs to.
        var log = new NullSimLog();
        _game = GameInitializer.Build(
            levelId, seed,
            configRoot: "",        // unused — catalogs pre-loaded via ResourceLoader
            _blockCatalog!,
            _relicCatalog!,
            _bonusCatalog,
            _profileStore,
            _dungeonStore,
            pid,
            log);
    }

    // Queue an input command (called immediately on user interaction).
    [JSExport]
    public static void EnqueueInput(string inputJson)
    {
        if (string.IsNullOrEmpty(inputJson)) return;
        var cmd = JsonSerializer.Deserialize<InputCommand>(inputJson);
        if (cmd != null) _inbox.Enqueue(cmd);
    }

    // Advance sim by one fixed step; drain the input queue first.
    // Returns the snapshot JSON. Called once per animation frame by WasmConnection.
    [JSExport]
    public static string Tick()
    {
        if (_game == null) return "{}";

        while (_inbox.TryDequeue(out var cmd)) ApplyInput(cmd);

        _game.Tick(_game.Config.FixedDt);
        _tick++;

        var snap = Snapshot.From(_game, _tick, _blockCache);
        return JsonSerializer.Serialize(snap);
    }

    [JSExport]
    public static void CloseSession() { _game = null; }

    private static void ApplyInput(InputCommand c)
    {
        switch (c.Kind)
        {
            case InputKind.PaddleX:        _game!.SetPaddleX(c.X); break;
            case InputKind.Serve:          _game!.Serve(); break;
            case InputKind.CastImbueIgnite:_game!.CastIgnite(); break;
            case InputKind.CastFireball:   _game!.CastFireball(); break;
            case InputKind.CastFireWall:   _game!.CastFireWall(); break;
            case InputKind.CastTurret:     _game!.CastTurret(); break;
            case InputKind.CastPhoenix:    _game!.CastPhoenix(); break;
            case InputKind.CastSlot:       _game!.CastSlot(c.Slot); break;
            case InputKind.Cheat:          _game!.ApplyCheat(c.Cheat ?? "", c.Value); break;
            case InputKind.RiftPick:       _game!.PickRiftModifier(c.RiftMod ?? ""); break;
        }
    }
}

internal sealed class NullSimLog : ISimLog
{
    public void Note(string cat, string msg, string? detail = null) { }
    public void Event(SimEvent e) { }
}
```

using System.Text.Json;
using Arkanoid.Core.Meta;

namespace ArkanoidWasm;

public static partial class MetaBridge
{
    private static readonly JsonSerializerOptions _opts = new() { PropertyNamingPolicy = JsonNamingPolicy.CamelCase };
    private static readonly WasmProfileStore _profileStore = GameBridge._profileStore; // shared singleton
    private static readonly WasmDungeonStore _dungeonStore = GameBridge._dungeonStore;

    // ── Profile ──────────────────────────────────────────────────────────────

    [JSExport]
    public static string GetProfile(string pid)
    {
        var p = _profileStore.Load(pid);
        return Serialize(new {
            level = p.Level, exp = p.Exp,
            sparks = p.Sparks, souls = p.Souls, insight = p.Insight,
            completedLevels = p.CompletedLevels,
            unlockedRelics = p.UnlockedRelics,
            spellLevels = p.SpellLevels,
            achievements = p.Achievements,

```typescript
// Loads the dotnet WASM module once; all other modules call wasmExports().
let _exports: Record<string, any> | null = null;

export async function initWasm(): Promise<void> {
  if (_exports) return; // already initialized

  // dotnet.js is served from /_framework/ (copied there by the build pipeline).
  const { dotnet } = await import("/_framework/dotnet.js" as any);

  const runtime = await dotnet
    .withApplicationArguments()
    .create();

  const cfg = runtime.getConfig();
  _exports = await runtime.getAssemblyExports(cfg.mainAssemblyName);

  // Initialize catalogs (reads embedded resources, caches in static fields).
  _exports!["ArkanoidWasm.GameBridge"].InitCatalogs();
}

export function wasmExports(): Record<string, any> {
  if (!_exports) throw new Error("WASM not initialized — call initWasm() first");
  return _exports;
}

export function gameBridge() { return wasmExports()["ArkanoidWasm.GameBridge"]; }
export function metaBridge() { return wasmExports()["ArkanoidWasm.MetaBridge"]; }
```

Mirrors the `Connection` class interface exactly. The Pixi ticker calls `Tick()` every frame.

```typescript
import type { Snapshot } from "./Connection";
import { gameBridge } from "../wasm/bridge";
import { log } from "../log";

export class WasmConnection {
  latest: Snapshot | null = null;
  onSnapshot: ((s: Snapshot) => void) | null = null;
  readonly runId: string;
  private _closed = false;
  private _ready = false;
  private _readyCbs: (() => void)[] = [];
  private _pendingInputs: object[] = [];
  private _rafId = 0;
  private _lastPhase = "";
  private _bridge = gameBridge();

  constructor(level: string, seed: number, runId: string, mode = "") {
    this.runId = runId;
    const pid = (typeof localStorage !== "undefined" && localStorage.getItem("ark_pid")) || "default";

    log("wasm", "initSession", { level, seed, runId, mode });
    this._bridge.InitSession(level, seed, runId, pid, mode);

    // Mark ready on next microtask so callers have time to set onSnapshot.
    Promise.resolve().then(() => {
      this._ready = true;
      this._readyCbs.forEach(cb => cb());
      this._readyCbs = [];
      this._startLoop();
    });
  }

  private _startLoop() {
    const tick = () => {
      if (this._closed) return;
      // Flush pending inputs.
      for (const cmd of this._pendingInputs) {
        this._bridge.EnqueueInput(JSON.stringify(cmd));
      }
      this._pendingInputs = [];

      const snapJson: string = this._bridge.Tick();
      const s = JSON.parse(snapJson) as Snapshot;
      this.latest = s;
      if (s.phase !== this._lastPhase) {
        log("phase", s.phase, { tick: s.tick, hp: s.hp });
        this._lastPhase = s.phase;
      }
      this.onSnapshot?.(s);
      this._rafId = requestAnimationFrame(tick);
    };
    this._rafId = requestAnimationFrame(tick);
  }

  private _send(cmd: object) {
    log("cmd", (cmd as any).kind, cmd);
    this._pendingInputs.push(cmd);
  }

  paddleX(x: number)         { this._send({ kind: "PaddleX", x }); }
  serve()                    { this._send({ kind: "Serve" }); }
  castIgnite()               { this._send({ kind: "CastImbueIgnite" }); }
  castFireball()             { this._send({ kind: "CastFireball" }); }
  castFireWall()             { this._send({ kind: "CastFireWall" }); }
  castTurret()               { this._send({ kind: "CastTurret" }); }
  castPhoenix()              { this._send({ kind: "CastPhoenix" }); }
  castSlot(slot: number)     { this._send({ kind: "CastSlot", slot }); }
  cheat(op: string, value = 0) { this._send({ kind: "Cheat", cheat: op, value }); }
  riftPick(id: string)       { this._send({ kind: "RiftPick", riftMod: id }); }

  close() {
    this._closed = true;
    cancelAnimationFrame(this._rafId);
    gameBridge().CloseSession();
  }

  whenReady(cb: () => void) {
    if (this._ready) cb(); else this._readyCbs.push(cb);
  }
}
```

- [ ] **Step 2: Create `frontend/public/sw.js`**

Cache-first strategy: the WASM bundle + assets download once, then the game works offline.

```javascript
const CACHE = "arkanoid-v1";

// On install: cache the WASM framework + all static assets.
self.addEventListener("install", (e) => {
  e.waitUntil(
    caches.open(CACHE).then((cache) =>
      cache.addAll(["/", "/index.html"])
        // Don't precache _framework files here — they're large and auto-fetched on first game load.
        // The browser caches them via normal HTTP caching.
    )
  );
  self.skipWaiting();
});

self.addEventListener("activate", (e) => {
  e.waitUntil(
    caches.keys().then((keys) =>
      Promise.all(keys.filter((k) => k !== CACHE).map((k) => caches.delete(k)))
    )
  );
  self.clients.claim();
});

self.addEventListener("fetch", (e) => {
  // Cache-first for _framework (WASM files) — they don't change between reloads.
  if (e.request.url.includes("/_framework/")) {
    e.respondWith(
      caches.open(CACHE).then((cache) =>
        cache.match(e.request).then(
          (cached) => cached ?? fetch(e.request).then((res) => { cache.put(e.request, res.clone()); return res; })
        )
      )
    );
    return;
  }
  // Network-first for everything else (index.html, assets).
  e.respondWith(fetch(e.request).catch(() => caches.match(e.request)));
});
```
